In binnaCustomer, you need to import Supplier + Inventory from binnaProviders and display them.
1. binnaCustomer/urls.py
from django.urls import path
from . import views

urlpatterns = [
    path('stores/', views.display_stores, name='display_stores'),
    path('stores/<int:supplier_id>/', views.store_details, name='store_details'),
    path('inventory/<int:inventory_id>/', views.inventory_details, name='inventory_details'),
]
2. binnaCustomer/views.py
from django.shortcuts import render, redirect
from binnaProviders.models import Supplier, Inventory, SupplierCategory, InventoryCategory, City


def display_stores(request):
    suppliers = Supplier.objects.get_all_suppliers()

    context = {
        'suppliers': suppliers,
        'cities': City.objects.get_all_cities(),
        'supplier_categories': SupplierCategory.objects.get_all_categories(),
    }

    return render(request, 'display_stores.html', context)


def store_details(request, supplier_id):
    supplier = Supplier.objects.get_supplier_by_id(supplier_id)
    inventory_items = Inventory.objects.get_supplier_inventory(supplier_id)

    context = {
        'supplier': supplier,
        'inventory_items': inventory_items,
    }

    return render(request, 'store_details.html', context)


def inventory_details(request, inventory_id):
    item = Inventory.objects.get_inventory_by_id(inventory_id)

    context = {
        'item': item,
        'supplier': item.supplier,
    }

    return render(request, 'inventory_details.html', context)

6. main project urls.py
Make sure your project urls.py includes binnaCustomer:
from django.urls import path, include
from django.conf import settings
from django.conf.urls.static import static

urlpatterns = [
    path('', include('binnaCustomer.urls')),
]

urlpatterns += static(settings.MEDIA_URL, document_root=settings.MEDIA_ROOT)
And in settings.py:
MEDIA_URL = '/static/'
MEDIA_ROOT = BASE_DIR / 'static'
This lets binnaCustomer display supplier and inventory images saved by binnaProviders.
